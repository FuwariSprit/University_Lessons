  char name[20];

  double weight;

    {"渡辺", 176.3, 65.6}
  };

}

void insertion(student students[]){
  int i, j;
  student tmp[1];
  for(i=1; i<NUM; i++){
    tmp[0] = students[i];
    for(j=i-1; j>=0; j--){
      if(students[j].height < tmp[0].height){
	students[j+1] = students[j];
      }
      else
	break;
    }
      students[j+1] = tmp[0];
  }
}
