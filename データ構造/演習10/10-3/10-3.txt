void swap(int *x, int *y){
  int tmp;

  tmp = *x;
  *x = *y;
  *y = tmp;
}

void quick(int x[], int left, int right){
  int pl = left;
  int pr = right;
  int pivot = x[(pl + pr) / 2];

  while(1){
    while(x[pl] < pivot){
      pl++;
    }
    while(x[pr] > pivot){
      pr--;
    }
    if(pr < pl){
      break;
    }

    swap(&x[pl], &x[pr]);
    pl++;
    pr--;
  }
  
  if(left < pr){
    quick(x, left, pr);
  }
  if(right > pl){
    quick(x, pl, right);
  }
}

int bin_search(int a[], int n, int key){
  int pl, pr, pc;
  
  pl = 0;
  pr = n-1;
  
  while(1){
    if(pr < pl){
      return -1;
    }
    
    pc = (pl+pr) / 2;
    
    if(a[pc] < key){
      pl = pc+1;
      bin_c++;
    }
    
    else if(key < a[pc]){
      pr = pc - 1;
      bin_c++;
    }
    
    else if(a[pc] == key){
      bin_c++;
      return pc;
    }
  }
}
