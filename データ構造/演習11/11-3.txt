int InsertNode(int no, int insdata, int data[]){
  int i;

  for(i=0; data[i]!=-1; i++){
    if(i>=NUM-1){
      return -1;
    }
  } 

  for(i=NUM; i>no; i--){
    data[i] = data[i-1];
  }

  data[no] = insdata;

  return 0;
}

int DeleteNode(int no, int data[]){

    return -1;
  }

    data[i] = data[i+1];
  }

  return 0;
}
